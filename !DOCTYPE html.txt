    <button onclick="shakeBall()">Shake the Ball</button>

    <script>
        const responses = [
            "it is certain",
            "diddy might be in your near future",
            "dont count on it",
            "reply hazzy, try again",
            "without a doubt",
            "Ask again later",
            "My reply is no",
            "Outlook good",
            "Very doubtful"
        ];

        function shakeBall() {
            const input = document.getElementById("questionInput");
            const ball = document.getElementById("ballGraphic");
            const answerDisplay = document.getElementById("answer");

            if (input.value.trim() === "") {
                alert("Please type a question first!");
                return;
            }

            answerDisplay.textContent = "...";
            ball.classList.add("shaking");

            setTimeout(() => {
                ball.classList.remove("shaking");
                const randomIndex = Math.floor(Math.random() * responses.length);
                answerDisplay.textContent = responses[randomIndex];
                input.value = ""; 
            }, 1000);
        }
    </script>
